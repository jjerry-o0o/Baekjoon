package test2;

import java.util.Scanner;

public class Test_2953 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int max = 0;
		int index = 0;
		for(int i=0; i<5; i++) {			
			int sum = 0;
			for(int j=0; j<4; j++) {
				int score = sc.nextInt();
				sum = sum + score;

				if(sum >= max) {
					max=sum;
					index = i+1;
			}
			}
		}
		System.out.println(index+" "+max);
	}
}
