import java.util.Scanner;

public class Test_9085 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int c = sc.nextInt();
		
		int sum = 0;
		for(int i=0; i<c; i++) {
			int n = sc.nextInt();
			for(int j=0; j<n; j++) {
				int x = sc.nextInt();
				sum = sum + x;
			}
			System.out.println(sum);
			sum=0;
		}
	}
}
