import java.util.Scanner;

public class Test_1934 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		
		for(int i=0; i<n; i++) {
			int a = sc.nextInt();
			int b = sc.nextInt();
			
			if(a>b) {
				int tmp = a;
				a = b;
				b = tmp;
			}
			
			int saveb = b;
			
			while(true) {
				if(b%a ==0) {
					break;
				}else {
					b = b+saveb;
				}
			}
			System.out.println(b);
		}
	}
}
