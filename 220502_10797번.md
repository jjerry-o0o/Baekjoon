import java.util.Scanner;

public class Test_10797 {
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int day = sc.nextInt();
		
		int[] carNum = new int[5];
		for(int i=0; i<5; i++) {
			carNum[i] = sc.nextInt();
		}
		
		int cnt = 0;
		for(int i=0; i<5; i++) {
			if(day == carNum[i]) {
				cnt++;
			}
		}
		System.out.println(cnt);
	}
}
