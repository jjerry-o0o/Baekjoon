import java.util.Scanner;

public class Test_4101 {

	public static void main(String[] args) {
		
		Scanner sc = new Scanner(System.in);
		
		while(true) {
			int one = sc.nextInt();
			int two = sc.nextInt();
			
			if(one == 0 && two == 0) {
				break;
			}

			if(one > two) {
				System.out.println("Yes");
			}else {
				System.out.println("No");
			}
		}
	}
}
