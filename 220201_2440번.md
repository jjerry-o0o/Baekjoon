import java.util.Scanner;

public class Test_2440 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int N = sc.nextInt();
		
		for (int line = 0; line < N; line++) {
			for (int i=1+line; i<=N; i++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
}
