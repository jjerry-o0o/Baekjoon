import java.util.Scanner;

public class Test_5596 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int S = 0;
		for(int i=0; i<4; i++) {
			int n = sc.nextInt();
			S += n;
		}
		
		int T = 0;
		for(int i=0; i<4; i++) {
			int n = sc.nextInt();
			T += n;
		}
		
		if(S > T) {
			System.out.println(S);
		}else if(S == T) {
			System.out.println(S);
		}else {
			System.out.println(T);
		}
	}
}
